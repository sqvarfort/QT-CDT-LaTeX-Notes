% definitions for the circuit elements

\def\gAxA{\gnqubit{H^{\otimes n}}{ddd}\w\A{gAxA}}
\def\gAxB{\gspace{H^{\otimes n}}\w\A{gAxB}}
\def\gAxC{\gspace{H^{\otimes n}}\w\A{gAxC}}
\def\gAxD{\gspace{H^{\otimes n}}\w\A{gAxD}}
\def\gBxA{\gnqubit{O_{f}}{dddd}\w\A{gBxA}}
\def\gBxB{\gspace{O_{f}}\w\A{gBxB}}
\def\gBxC{\gspace{O_{f}}\w\A{gBxC}}
\def\gBxD{\gspace{O_{f}}\w\A{gBxD}}
\def\gBxE{\gspace{O_{f}}\w\A{gBxE}}
\def\gCxA{\gnqubit{H^{\otimes n}}{ddd}\w\A{gCxA}}
\def\gCxB{\gspace{H^{\otimes n}}\w\A{gCxB}}
\def\gCxC{\gspace{H^{\otimes n}}\w\A{gCxC}}
\def\gCxD{\gspace{H^{\otimes n}}\w\A{gCxD}}
\def\gDxA{\gnqubit{\mathcal{M}}{ddd}\w\A{gDxA}}
\def\gDxB{\gspace{\mathcal{M}}\w\A{gDxB}}
